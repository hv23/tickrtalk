295     def self.build_multipart_bodies(parts)
 
296       boundary = Time.now.to_i.to_s(16)
 
297       body = ""
 
298       parts.each do |key, value|
 
299         esc_key = CGI.escape(key.to_s)
 
300         body << "--#{boundary}#{CRLF}"
 
301         if value.respond_to?(:read)
 
302           body << "Content-Disposition: form-data; name=\"#{esc_key}\"; filename=\"#{File.basename(value.path)}\"#{CRLF}"
 
303           body << "Content-Type: #{mime_type(value.path)}#{CRLF*2}"
 
304           body << value.read
 
305         else
 
306           body << "Content-Disposition: form-data; name=\"#{esc_key}\"#{CRLF*2}#{value}"
 
307         end
 
308         body << CRLF
 
309       end
 
310       body << "--#{boundary}--#{CRLF*2}"
 
311       {
 
312         :body => body,
 
313         :headers => {"Content-Type" => "multipart/form-data; boundary=#{boundary}"}
 
314       }
 
315     end
